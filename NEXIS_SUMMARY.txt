ПРИМЕР 1: Создание бота
───────────────────────────────────────────────────────────────────────
const response = await fetch('/api/bots', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    name: 'NewsBot',
    description: 'Получайте новости',
    commands: [
      { cmd: '/start', description: 'Начать' },
      { cmd: '/latest', description: 'Последние новости' }
    ]
  })
});

ПРИМЕР 2: Подписка на бота
───────────────────────────────────────────────────────────────────────
await fetch('/api/bots/1/subscribe', { method: 'POST' });

ПРИМЕР 3: Отправка команды
───────────────────────────────────────────────────────────────────────
await fetch('/api/bots/1/command', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ command: '/start' })
});

ПРИМЕР 4: Слушание сообщений от ботов
───────────────────────────────────────────────────────────────────────
socket.on('message-from-bot', (data) => {
  console.log(`Бот ${data.botName}: ${data.text}`);
});
